import { useState } from 'react'
 
export function Counter({ children }) {
    const [count, setCount] = useState(0)
    return (
        <button className="text-3xl font-bold underline" onClick={() => setCount(count + 1)}>
            {children}
            {count}
        </button>
    )
}
