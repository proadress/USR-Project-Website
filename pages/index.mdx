# Buttons
import {useState} from 'react';

export function Counter({children}) {
    const [count, setCount] = useState(0);
    return (
    <button onClick={()=> setCount(count+1)}
        className="text-blue-100">
        {children} {count} 
    </button>)
}


<Counter>**Clicks:** </Counter>
